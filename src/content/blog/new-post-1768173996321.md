import asyncio
from datetime import datetime
from typing import List, Optional

class SyntaxTester:
    """
    A multi-purpose class to verify syntax highlighting for 2026 themes.
    Includes decorators, type hints, and f-strings.
    """
    VERSION = "2.0.1"

    def __init__(self, theme_name: str = "Modern Dark"):
        self.theme = theme_name
        self.created_at = datetime.now()
        self.is_active = True

    @staticmethod
    def get_timestamp() -> str:
        return datetime.utcnow().strftime("%Y-%m-%d %H:%M:%S")

    async def process_data(self, items: List[int]) -> Optional[float]:
        try:
            print(f"[{self.get_timestamp()}] Processing {len(items)} items...")
            
            # Simulate a mathematical operation with various operators
            result = sum([x**2 for x in items if x > 0])
            
            await asyncio.sleep(0.5) # Test async highlighting
            
            if result == 0:
                raise ValueError("Calculation resulted in zero.")
                
            return float(result / len(items))
            
        except Exception as e:
            print(f"Error detected: {e}")
            return None

# Instantiate and run the test
if __name__ == "__main__":
    tester = SyntaxTester(theme_name="OLED_High_Contrast")
    data_points = [10, 20, 30, 40]
    
    # Running the async function
    loop = asyncio.get_event_loop()
    final_score = loop.run_until_complete(tester.process_data(data_points))
    
    print(f"Final Score: {final_score}")
